import javax.swing.*; import java.awt.event.ActionEvent; import java.awt.event.ActionListener; import java.util.HashMap; import java.util.Map; public class HVACSystemCalculatorGUI { public static void main(String[] args) { JFrame frame = new JFrame("HVAC System Calculator"); frame.setSize(400, 400); frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); frame.setLayout(null); JLabel labelRefrigerant = new JLabel("Select Refrigerant:"); labelRefrigerant.setBounds(20, 20, 150, 25); frame.add(labelRefrigerant); String[] refrigerants = {"R22", "R410a", "R454b", "R32"}; JComboBox refrigerantBox = new JComboBox<>(refrigerants); refrigerantBox.setBounds(180, 20, 150, 25); frame.add(refrigerantBox); JLabel labelPounds = new JLabel("Enter Quantity (lbs):"); labelPounds.setBounds(20, 60, 150, 25); frame.add(labelPounds); JTextField textPounds = new JTextField(); textPounds.setBounds(180, 60, 150, 25); frame.add(textPounds); JLabel labelService = new JLabel("Select Service:"); labelService.setBounds(20, 100, 150, 25); frame.add(labelService); String[] services = {"installation", "repair", "maintenance"}; JComboBox serviceBox = new JComboBox<>(services); serviceBox.setBounds(180, 100, 150, 25); frame.add(serviceBox); JButton calculateBtn = new JButton("Calculate"); calculateBtn.setBounds(130, 150, 120, 30); frame.add(calculateBtn); JTextArea resultArea = new JTextArea(); resultArea.setBounds(20, 200, 340, 120); resultArea.setEditable(false); frame.add(resultArea); calculateBtn.addActionListener(new ActionListener() { @Override public void actionPerformed(ActionEvent e) { Map refrigerantPrices = new HashMap<>(); refrigerantPrices.put("R22", 250.0); refrigerantPrices.put("R410a", 150.0); refrigerantPrices.put("R454b", 99.0); refrigerantPrices.put("R32", 89.0); Map serviceCosts = new HashMap<>(); serviceCosts.put("installation", 500); serviceCosts.put("repair", 200); serviceCosts.put("maintenance", 100); try { String refrigerant = (String) refrigerantBox.getSelectedItem(); double pounds = Double.parseDouble(textPounds.getText()); String service = (String) serviceBox.getSelectedItem(); double refrigerantCost = refrigerantPrices.getOrDefault(refrigerant, 0.0) * pounds; int serviceCost = serviceCosts.getOrDefault(service, 0); double total = refrigerantCost + serviceCost; resultArea.setText("--- Summary ---\n"); resultArea.append("Refrigerant: " + refrigerant + "\n"); resultArea.append("Quantity: " + pounds + " lbs\n"); resultArea.append("Refrigerant Cost: $" + refrigerantCost + "\n"); resultArea.append("Service: " + service + "\n"); resultArea.append("Service Cost: $" + serviceCost + "\n"); resultArea.append("Total Cost: $" + total); } catch (NumberFormatException ex) { resultArea.setText("Please enter a valid number for pounds."); } } }); frame.setVisible(true); } }
